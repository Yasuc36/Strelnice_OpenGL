107 unsigned int max;
108 max = 1;

167bool GameObject::LoadSprite(const std::string path, unsigned int numImages, unsigned int numColumns)
168{
169 // SETUP PHASE
170 UseBlend = true;
171 BlendFirst = GL_SRC_ALPHA;
172 BlendSecond = GL_ONE_MINUS_SRC_ALPHA;
173 NumImages = numImages;
174 NumColumns = numColumns;
175 std::vector<float> verticesUv =
176 {
177 // x y z u v
178 -0.5f, -0.5f, 0.0f, 0.0f, 0.0f,
179 -0.5f, 0.5f, 0.0f, 0.0f, 1.0f,
180 0.5f, -0.5f, 0.0f, 1.0f, 0.0f,
181 0.5f, 0.5f, 0.0f, 1.0f, 1.0f
182 };
183 std::vector<unsigned int> indices =
184 {
185 0, 1, 2,
186 1, 2, 3
187 };
188 NumVertices = verticesUv.size() / 5;
189 NumTriangles = indices.size() / 3;
190
191 // Texture
192 Texture = pgr::createTexture(path);
193 // abort if the loader fails
194 if (Texture == 0)
195 {
196 std::cout << "Could'nt load texture " << path << ".\n";
197 return false;
198 }
199 CHECK_GL_ERROR();
200
201 // Buffers
202 Vao.ConnectBuffers(Vbo.Index, Ebo.Index);
203 Vbo.SetData(verticesUv);
204 Ebo.SetData(indices);
205 CHECK_GL_ERROR();
206
207 // Shaders
208 glBindVertexArray(Vao.Index);
209 glEnableVertexAttribArray(Shaders->Pos);
210 glVertexAttribPointer(Shaders->Pos, 3, GL_FLOAT, GL_FALSE, 5 * sizeof(float), 0); // Stride (xyzuv)
211 glEnableVertexAttribArray(Shaders->TexCoord);
212 glVertexAttribPointer(Shaders->TexCoord, 2, GL_FLOAT, GL_FALSE, 5 * sizeof(float), (void*)(3 * sizeof(float))); // Stride (xyzuv) + offset (xyz)
213 CHECK_GL_ERROR();
214
215 // EXIT PHASE
216 glBindVertexArray(0);
217 UseTexture = true;
218 Initialized = true;
219 CHECK_GL_ERROR();
220 return true;
221}

279void GameObject::Draw(const glm::mat4& viewMatrix, const glm::mat4& projectionMatrix, const int activeCamera)
280{
281 if (Initialized)
282 {
283 if (UseBlend)
284 {
285 glEnable(GL_BLEND);
286 glBlendFunc(BlendFirst, BlendSecond);
287 }
288 for ( auto & it : Instances) // For each instance:
289 {
290 glStencilFunc(GL_ALWAYS, it.StencilId, 0);
291 if ((it.Ui && activeCamera != 0) || !it.Draw) continue; // Skip UI for spectate cameras and object, that shouldn't be drawn
292 glm::mat4 view;
293 glm::mat4 projection;
294 if (it.Ui) view = projection = glm::mat4(1.0f); // Set different view projection matrix for UI objects
295 else
296 {
297 view = viewMatrix;
298 projection = projectionMatrix;
299 }
300 if (Shaders != nullptr)
301 {
302 // Set correct shader programs and locations
303 glUseProgram(Shaders->Index);
304 glUniformMatrix4fv(Shaders->ViewMat, 1, GL_FALSE, glm::value_ptr(view));
305 glUniformMatrix4fv(Shaders->ProjectionMat, 1, GL_FALSE, glm::value_ptr(projection));
306 glUniformMatrix4fv(Shaders->ModelMat, 1, GL_FALSE, glm::value_ptr(it.ModelMatrix));
307 if(it.OwnColor) glVertexAttrib3f(Shaders->Color, it.BaseColor.r, it.BaseColor.g, it.BaseColor.b);
308 else glVertexAttrib3f(Shaders->Color, BaseColor.r, BaseColor.g, BaseColor.b);
309 glVertexAttrib3f(Shaders->Pos, it.Position.x, it.Position.y, it.Position.z);
310
311 //material
312 glUniform3f(Shaders->MaterialAmbient, ObjectMaterial.ambient.x, ObjectMaterial.ambient.y, ObjectMaterial.ambient.z);
313 glUniform3f(Shaders->MaterialDiffuse, ObjectMaterial.diffuse.x, ObjectMaterial.diffuse.y, ObjectMaterial.diffuse.z);
314 glUniform3f(Shaders->MaterialSpecular, ObjectMaterial.specular.x, ObjectMaterial.specular.y, ObjectMaterial.specular.z);
315 glUniform1f(Shaders->MaterialShininess, ObjectMaterial.shininess);
316
317 // textures
318 glUniform1i(Shaders->IsUi, it.Ui);
319 if(!HaveNormals) glVertexAttrib3f(Shaders->Normal, 0.0f, 0.0f, 0.0f);
320 glUniform1i(Shaders->TexSampler, 0);
321 glUniform1i(Shaders->UseTexture, UseTexture);
322 glActiveTexture(GL_TEXTURE0 + 0);
323 glBindTexture(GL_TEXTURE_2D, Texture);
324
325 // texture animation
326 glUniform1i(Shaders->NumImages, NumImages);
327 glUniform1i(Shaders->NumColumns, NumColumns);
328 glUniform1i(Shaders->CurrImage, it.CurrImage);
329
330 // texture shift
331 glUniform2f(Shaders->Shift, it.Shift.x, it.Shift.y);
332
333 glBindVertexArray(Vao.Index);
334 if (NumFrames > 1) glVertexAttribPointer(Shaders->Pos, 3, GL_FLOAT, GL_FALSE, 0, (void*)(it.CurrFrame * (NumVertices / NumFrames) * 3 * sizeof(float)));
335 if (UseEBO) glDrawElements(GL_TRIANGLES, NumTriangles * 3, GL_UNSIGNED_INT, 0);
336 else glDrawArrays(GL_TRIANGLES, 0, NumTriangles * 3);
337 glBindVertexArray(0);
338 }
339 }
340 if (UseBlend) glDisable(GL_BLEND); // Disable blending, if it was used
341 }
342 else std::cout << "Can't draw Object: object not initialized properly!" << std::endl;
343}

346void GameObject::Update(const float elapsedTime, const float deltaTime)
347{
348 for (auto& it : Instances)

356bool GameObject::GetCollision(glm::vec3 position, float radius)
357{
358 for (int i = 0; i < Instances.size(); i++)
359 {
360 glm::vec3 thisPos = glm::vec3(Instances[i].Position.x, 0.0f, Instances[i].Position.z);
361 glm::vec3 otherPos = glm::vec3(position.x, 0.0f, position.z);
362 float distance = glm::distance(thisPos, otherPos);
363 if (distance <= (Radius * Instances[i].Scale.x + radius) ||
364 distance <= (Radius * Instances[i].Scale.y + radius) ||
365 distance <= (Radius * Instances[i].Scale.z + radius)) return true;
366 }
367 return false;
