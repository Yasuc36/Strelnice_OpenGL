155 glUniform1i(commonShaders.SpotLightUse, lights[spotLightPos].Use);

157
158 glUniform3f(commonShaders.ViewPos, cameras[activeCamera].Position.x, cameras[activeCamera].Position.y, cameras[activeCamera].Position.z);

162 glUniform1f(commonShaders.FogDensity, fogDensity*fogMultiplier);

166 // fog color

169 glUniform1f(colorShaders.FogDensity, fogDensity);

174
175 for (auto& it : objects) // First draw objects, that doesn't use blending
176 {
177 if (!it.UseBlend)
178 {
179 if (it.Name != "Skybox") it.Draw(view, projection, activeCamera);
180 else it.Draw(glm::lookAt(glm::vec3(0.0f), glm::vec3(0.0f) + cameras[activeCamera].Front, cameras[activeCamera].Up), projection, activeCamera);
181 }
182 }
183 for (auto& it : objects) // Then draw objects, that does use blending
184 {
185 if (it.UseBlend)
186 {
187 if (it.Name == "GrassStems" || it.Name == "Fire")
188 {
189 for (auto& i : it.Instances) i.Transform(i.Position, glm::vec3(i.Rotation.x, 90.0f - cameras[activeCamera].Yaw, i.Rotation.z), i.Scale);
190 }
191 it.Draw(view, projection, activeCamera);
192 }
